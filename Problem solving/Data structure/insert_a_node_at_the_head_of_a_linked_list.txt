/*
 * For your reference:
 *
 * SinglyLinkedListNode {
 *     int data;
 *     SinglyLinkedListNode* next;
 * };
 *
 */
SinglyLinkedListNode* insertNodeAtHead(SinglyLinkedListNode* llist, int data) 
{
    SinglyLinkedListNode* head=new SinglyLinkedListNode(data);
    head->next=llist;
    return head;
}
